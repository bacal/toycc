#+begin_src shell

** Run
#+begin_src shell
  cargo run -- <arguments>
#+end_src
or
#+begin_src shell
  ./target/release/toycc <arguments>
#+end_src

** Clean
#+begin_src shell
  cargo clean
#+end_src
